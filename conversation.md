Step 2 — Create Server

server.js

const express = require("express")
const WebSocket = require("ws")
const pty = require("node-pty")
const cors = require("cors")

const app = express()
app.use(cors())

const server = app.listen(7777, () => {
  console.log("Sidecar running on port 7777")
})

const wss = new WebSocket.Server({ server })

wss.on("connection", (ws) => {

  const shell = pty.spawn("bash", [], {
    name: "xterm-color",
    cols: 80,
    rows: 30,
    cwd: process.env.HOME,
    env: process.env
  })

  shell.onData(data => ws.send(data))

  ws.on("message", msg => shell.write(msg))
})

Now you have a WebSocket terminal server.
Step 3 — Filesystem API

Add to server.js:

const fs = require("fs")
const path = require("path")

app.get("/fs/list", (req, res) => {

  const dir = req.query.dir || process.cwd()

  fs.readdir(dir, { withFileTypes: true }, (err, files) => {

    if (err) return res.status(500).send(err)

    const result = files.map(f => ({
      name: f.name,
      type: f.isDirectory() ? "dir" : "file"
    }))

    res.json(result)
  })
})

Now the browser can browse files.

const express = require("express")
const WebSocket = require("ws")
const pty = require("node-pty")
const Docker = require("dockerode")
const cors = require("cors")
const fs = require("fs")
const path = require("path")

const app = express()
app.use(cors())

const docker = new Docker()

const server = app.listen(7777, () => {
  console.log("Sidecar agent running on port 7777")
})

const wss = new WebSocket.Server({ server })

// Terminal websocket
wss.on("connection", (ws) => {

  const shell = pty.spawn("bash", [], {
    name: "xterm-color",
    cols: 80,
    rows: 30,
    cwd: process.env.HOME,
    env: process.env
  })

  shell.onData(data => {
    ws.send(data)
  })

  ws.on("message", msg => {
    shell.write(msg)
  })

})

// Filesystem API
app.get("/fs/list", async (req, res) => {

  const dir = req.query.dir || process.cwd()

  try {

    const files = await fs.promises.readdir(dir, { withFileTypes: true })

    const result = files.map(f => ({
      name: f.name,
      type: f.isDirectory() ? "dir" : "file"
    }))

    res.json(result)

  } catch (err) {

    res.status(500).json({ error: err.message })

  }

})

// Docker containers
app.get("/docker/containers", async (req, res) => {

  try {

    const containers = await docker.listContainers()

    res.json(containers)

  } catch (err) {

    res.status(500).json({ error: err.message })

  }

})

import { useEffect, useState } from "react"

export default function FileExplorer() {

  const [path, setPath] = useState("/")
  const [files, setFiles] = useState([])

  async function load(dir) {

    const res = await fetch(`http://localhost:7777/fs/list?dir=${dir}`)
    const data = await res.json()

    setFiles(data)
    setPath(dir)

  }

  useEffect(() => {
    load("/")
  }, [])

  function openFile(file) {

    if (file.type === "dir") {
      const newPath = path === "/" ? `/${file.name}` : `${path}/${file.name}`
      load(newPath)
    }

  }

  return (

    <div style={{ padding:"10px", color:"white" }}>

      <div style={{ marginBottom:"10px" }}>
        Path: {path}
      </div>

      {files.map((f, i) => (

        <div
          key={i}
          onClick={() => openFile(f)}
          style={{
            cursor:"pointer",
            padding:"4px"
          }}
        >

          {f.type === "dir" ? "📁" : "📄"} {f.name}

        </div>

      ))}

    </div>

  )

}

1. Agent — Log Streaming Endpoint

We will stream logs over WebSockets using tail -f.

Edit:

agent/server.js

Add this section.

const { spawn } = require("child_process")

wss.on("connection", (ws, req) => {

  const url = req.url || ""

  if (url.startsWith("/logs")) {

    const params = new URLSearchParams(url.split("?")[1])
    const file = params.get("file") || "/var/log/syslog"

    const tail = spawn("tail", ["-f", file])

    tail.stdout.on("data", data => {
      ws.send(data.toString())
    })

    tail.stderr.on("data", data => {
      ws.send(data.toString())
    })

    ws.on("close", () => {
      tail.kill()
    })

  }

})

Now the agent supports:

ws://server:7777/logs?file=/var/log/syslog

and streams log lines continuously.
2. Log Viewer UI Component

import { useEffect, useRef } from "react"

export default function LogViewer({ machine }) {

  const ref = useRef()

  useEffect(() => {

    const ws = new WebSocket(
      machine.url.replace("http", "ws") + "/logs?file=/var/log/syslog"
    )

    ws.onmessage = (event) => {

      const line = document.createElement("div")
      line.textContent = event.data

      ref.current.appendChild(line)
      ref.current.scrollTop = ref.current.scrollHeight

    }

    return () => ws.close()

  }, [machine])

  return (

    <div
      ref={ref}
      style={{
        height: "100%",
        overflow: "auto",
        background: "#111",
        color: "#0f0",
        fontFamily: "monospace",
        padding: "10px"
      }}
    />

  )

}

This renders a real-time scrolling log window.

import { useEffect, useState } from "react"

export default function FileExplorer({ machine }) {

  const [path, setPath] = useState("/")
  const [files, setFiles] = useState([])

  async function load(dir) {

    const res = await fetch(`${machine.url}/fs/list?dir=${dir}`)
    const data = await res.json()

    setFiles(data)
    setPath(dir)

  }

  useEffect(() => {
    load("/")
  }, [machine])

  function open(item) {

    if (item.type === "dir") {

      const newPath =
        path === "/" ? `/${item.name}` : `${path}/${item.name}`

      load(newPath)

    }

  }

  return (

    <div style={{ padding: "10px", color: "white" }}>

      <div style={{ marginBottom: "10px" }}>
        Path: {path}
      </div>

      {files.map((f, i) => (

        <div
          key={i}
          onClick={() => open(f)}
          style={{
            cursor: "pointer",
            padding: "4px"
          }}

          {f.type === "dir" ? "📁" : "📄"} {f.name}

function loadPlugins(app) {

  const dir = path.join(__dirname, "plugins")

  if (!fs.existsSync(dir)) return

  const files = fs.readdirSync(dir)

  files.forEach(file => {
